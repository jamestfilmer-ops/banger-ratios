BLOCK 2 — Create the follows table
This is what powers finding and following friends. One row per follow relationship.
-- BLOCK 2: Create follows table for social features
CREATE TABLE IF NOT EXISTS follows (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  follower_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  following_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  created_at timestamptz DEFAULT now(),
  UNIQUE(follower_id, following_id)
);

-- Enable fuzzy username search
CREATE EXTENSION IF NOT EXISTS pg_trgm;
CREATE INDEX IF NOT EXISTS profiles_username_trgm
  ON profiles USING gin(username gin_trgm_ops);

-- Row Level Security for follows
ALTER TABLE follows ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "read follows" ON follows;
CREATE POLICY "read follows" ON follows FOR SELECT USING (true);
DROP POLICY IF EXISTS "own follows" ON follows;
CREATE POLICY "own follows" ON follows FOR ALL
  USING (auth.uid() = follower_id)
  WITH CHECK (auth.uid() = follower_id);

 SECTION 4: Security & Privacy
This section explains the security system that protects your users' data, how to harden it further, and the code changes needed to enforce strong passwords client-side.

How Your Security System Works
Banger Ratios uses three layers of security that work together:

Layer 1 — Row Level Security (RLS) in Supabase
Every table in your database has RLS enabled. This means Supabase automatically enforces who can read or write each row — even if someone gets your anon key, they cannot read another user's private data or write to someone else's profile.

Table	Who Can Read / Write
albums	Anyone can read. Authenticated users can write (to save new albums when rating).
tracks	Anyone can read. Authenticated users can write.
ratings	Anyone can read. Only the owner (auth.uid() = user_id) can write.
profiles	Anyone can read. Only the owner (auth.uid() = id) can write.
follows	Anyone can read. Only the follower (auth.uid() = follower_id) can write.
storage.objects	Anyone can read public buckets. Only the file owner can upload/delete.

Layer 2 — Supabase Auth
Authentication is handled entirely by Supabase Auth. This means: passwords are hashed and never stored in plaintext, sessions use short-lived JWT tokens, and email confirmation prevents fake accounts. You never see or store passwords — Supabase handles it all.

Layer 3 — Client-side validation
Before any data reaches Supabase, the frontend validates it. This catches errors early and provides clear feedback to users. The updated auth page (below) enforces strong passwords before the signup request is sent.

Security Improvements to Make

1. Strong Password Enforcement — auth/page.js
Replace the password validation in auth/page.js. The new version requires: minimum 8 characters, at least one uppercase letter, at least one number, and at least one special character. It shows the user exactly what is missing as they type.

// Add this validator function BEFORE your Auth() component:
function validatePassword(pw) {
  const errors = []
  if (pw.length < 8) errors.push('At least 8 characters')
  if (!/[A-Z]/.test(pw)) errors.push('At least one uppercase letter')
  if (!/[0-9]/.test(pw)) errors.push('At least one number')
  if (!/[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(pw))
    errors.push('At least one special character (!@#$...)')
  return errors
}

// Replace your signUp function with this:
const signUp = async () => {
  if (!email || !password) { setMessage('Please enter email and password.'); return }
  const pwErrors = validatePassword(password)
  if (pwErrors.length > 0) {
    setMessage('Password must have: ' + pwErrors.join(', '))
    return
  }
  setLoading(true)
  const { data, error } = await supabase.auth.signUp({ email, password })
  if (error) { setMessage(error.message); setLoading(false); return }
  if (data?.user) {
    await supabase.from('profiles').upsert({
      id: data.user.id,
      username: username || email.split('@')[0],
      email,
    }, { onConflict: 'id', ignoreDuplicates: true })
  }
  setMessage('Account created! Check your email to confirm, then sign in.')
  setLoading(false)
}

export default function Friends() {

  const [feed, setFeed] = useState([])
  const [me, setMe] = useState(null)

  useEffect(() => {
    async function init() {
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) { router.push('/auth'); return }
      setMe(user)
      const { data: follows } = await supabase
        .from('follows')
        .select('following_id')
        .eq('follower_id', user.id)
      const ids = (follows || []).map(f => f.following_id)
      setFollowing(new Set(ids))
      if (ids.length > 0) {
        const { data: activity } = await supabase
          .from('ratings')
          .select(`score, created_at,
            profiles ( username ),
            tracks ( name ),
            albums ( name, artist_name, artwork_url, itunes_collection_id )`)
          .in('user_id', ids)
          .order('created_at', { ascending: false })
          .limit(20)
        setFeed(activity || [])
      }
      setLoading(false)
    }
    init()
  }, [])

  async function searchUsers(q) {
    if (!q.trim()) { setResults([]); return }

      .from('profiles')
      .select('id, username, email')

        .insert({ follower_id: me.id, following_id: userId })
      setFollowing(prev => new Set([...prev, userId]))

async function uploadFile(file, bucket, pathSuffix) {
  const ext = file.name.split('.').pop()
  const path = `${user.id}/${pathSuffix}.${ext}`
  const { error } = await supabase.storage
    .from(bucket).upload(path, file, { upsert: true })
  if (error) { console.error(error); return null }
  const { data: { publicUrl } } = supabase.storage
    .from(bucket).getPublicUrl(path)
  return publicUrl
}

async function uploadBanner(file) {
  const url = await uploadFile(file, 'banners', 'banner')
  if (!url) return
  await supabase.from('profiles').update({ banner_url: url }).eq('id', user.id)
  setProfile(prev => ({ ...prev, banner_url: url }))
}

async function uploadAvatar(file) {
  const url = await uploadFile(file, 'avatars', 'avatar')
  if (!url) return
  await supabase.from('profiles').update({ avatar_url: url }).eq('id', user.id)
  setProfile(prev => ({ ...prev, avatar_url: url }))
}

async function connectSpotify() {
  await supabase.auth.signInWithOAuth({
    provider: 'spotify',
    options: {
      scopes: 'user-read-recently-played user-top-read user-read-private',
      redirectTo: `${window.location.origin}/profile`,
    }
  })
}
