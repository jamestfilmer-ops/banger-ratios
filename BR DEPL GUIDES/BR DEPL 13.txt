ROOT = os.path.dirname(os.path.abspath(__file__))
changed = []
skipped = []

def read(path):
    full = os.path.join(ROOT, path)
    if not os.path.exists(full):
        return None
    with open(full) as f:
        return f.read()
 
def write(path, content):
    full = os.path.join(ROOT, path)
    os.makedirs(os.path.dirname(full), exist_ok=True)
    with open(full, 'w') as f:
        f.write(content)
    changed.append(path)
    print(f'  ✓  {path}')
 
def patch(path, old, new, description=''):
    content = read(path)
    if content is None:
        errors.append(f'{path}: file not found')
        return False
    if old not in content:
        skipped.append(f'{path}: target not found — {description or old[:60]}')
        return False
    if new in content and old not in content:
        skipped.append(f'{path}: already patched')
        return False
    write(path, content.replace(old, new, 1))
    return True

  const [displayName, setDisplayName] = useState('')
  const [bio, setBio] = useState('')
  const [spotifyUrl, setSpotifyUrl] = useState('')
  const [appleMusicUrl, setAppleMusicUrl] = useState('')
  const [favAlbums, setFavAlbums] = useState(['','','','',''])

  const [avatarUrl, setAvatarUrl] = useState('')

    supabase.auth.getUser().then(async ({ data }) => {
      if (!data?.user) { router.push('/auth'); return }
      setUser(data.user)
      const { data: p } = await supabase.from('profiles').select('*').eq('id', data.user.id).single()
      if (p) {
        setProfile(p)
        setUsername(p.username || '')
        setDisplayName(p.display_name || '')
        setBio(p.bio || '')
        setSpotifyUrl(p.spotify_url || '')
        setAppleMusicUrl(p.apple_music_url || '')
        setAvatarUrl(p.avatar_url || '')

          </label>
        </div>
      </div>
 
      <div style={CARD}>
        <h2 style={{ marginTop:0 }}>Profile Info</h2>
        <label style={LBL}>USERNAME</label>
        <input style={INP} value={username} onChange={e=>setUsername(e.target.value)} placeholder='username' />
        <label style={LBL}>DISPLAY NAME</label>
        <input style={INP} value={displayName} onChange={e=>setDisplayName(e.target.value)} placeholder='Display Name' />
        <label style={LBL}>BIO</label>

        <button style={BTN} onClick={saveProfile}>Save Profile</button>
      </div>
 
      <div style={CARD}>
        <h2 style={{ marginTop:0 }}>Favorite Albums</h2>
        <p style={{ color:'#888', fontSize:'0.85rem' }}>Appears at the top of your profile.</p>
        {favAlbums.map((a,i) => (
          <div key={i} style={{ display:'flex', alignItems:'center', gap:'0.5rem', marginBottom:'0.4rem' }}>

              onChange={e=>{const arr=[...favAlbums];arr[i]=e.target.value;setFavAlbums(arr)}}

// File: src/middleware.js
import { NextResponse } from 'next/server'
const rateMap = new Map()
export function middleware(request) {
  const ip = request.headers.get('x-forwarded-for') || 'unknown'
  const now = Date.now()
  const windowMs = 60 * 1000
  const maxRequests = 60
  const entry = rateMap.get(ip) || { count: 0, start: now }
  if (now - entry.start > windowMs) { entry.count = 1; entry.start = now }
  else { entry.count++ }
  rateMap.set(ip, entry)
  if (rateMap.size > 5000) {
    for (const [k, v] of rateMap) { if (now - v.start > windowMs * 2) rateMap.delete(k) }
  }
  if (entry.count > maxRequests) {
    return new NextResponse('Too many requests', { status: 429, headers: { 'Retry-After': '60' } })
  }
  return NextResponse.next()
}
export const config = { matcher: ['/api/:path*', '/auth/:path*'] }

SECTION 3  —  THE ZSH: EVENT NOT FOUND ERROR — EXPLAINED

zsh: event not found: id)   means zsh (your shell) saw the ! or backtick characters inside your grep pattern and tried to interpret them as shell history commands. This is a zsh-specific behaviour.

Why it happens
In zsh, ! inside double-quoted strings triggers history expansion. So grep -n 'emoji|🎵' works fine but any pattern with special characters can trigger it. The autopatcher in Section 2 avoids this entirely because Python does not have this problem.

If you need to run grep manually in zsh
Use single quotes	grep -n 'pattern' file.js   (single quotes are safe)
Escape the !	grep -n "pattern\!" file.js
Disable history expansion	set +H   (run this once, lasts for the session)
Use Python instead	python3 -c "import re,sys; [print(i+1,l.rstrip()) for i,l in enumerate(open('file.js')) if 'pattern' in l]"
