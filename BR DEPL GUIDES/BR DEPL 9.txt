 SECTION 1 — Fix Your Database (Run SQL First)
Run every block below in Supabase → SQL Editor → New query → paste → Run → wait for Success. Run them in order. All are safe to re-run if you have already run some.
WHY FIRST: The 'Database error saving new user' crash is caused by a mismatch between the handle_new_user trigger and your profiles table. Block 1 below fixes it. Nothing else will work correctly until this runs.

BLOCK 1 — Fix the signup trigger (fixes 'Database error saving new user')
The existing trigger tries to insert into profiles but the table schema changed when new columns were added. This replaces it with a version that handles all columns safely.
SQL → SUPABASE SQL EDITOR → RUN
-- BLOCK 1: Fix handle_new_user trigger
-- This fixes the 'Database error saving new user' crash on signup.
-- Safe to re-run. Replaces the old trigger with a correct version.

CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  INSERT INTO profiles (
    id, email, username, created_at,
    banner_url, avatar_url, spotify_id, bio, display_name
  )
  VALUES (
    new.id,
    new.email,
    split_part(new.email, '@', 1),
    now(),
    null, null, null, null, null
  )
  ON CONFLICT (id) DO NOTHING;
  RETURN new;
END;
$$;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

BLOCK 2 — Add missing profile columns
Adds banner_url, avatar_url, spotify_id, bio, and display_name to the profiles table. Safe to run even if some already exist.
SQL → SUPABASE SQL EDITOR → RUN
-- BLOCK 2: Add social profile columns
DO $$ BEGIN
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='banner_url')
    THEN ALTER TABLE profiles ADD COLUMN banner_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='avatar_url')
    THEN ALTER TABLE profiles ADD COLUMN avatar_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='spotify_id')
    THEN ALTER TABLE profiles ADD COLUMN spotify_id text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='bio')
    THEN ALTER TABLE profiles ADD COLUMN bio text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='display_name')
    THEN ALTER TABLE profiles ADD COLUMN display_name text; END IF;
END $$;

BLOCK 3 — Create the follows table
SQL → SUPABASE SQL EDITOR → RUN
-- BLOCK 3: follows table for friend/follow features
CREATE TABLE IF NOT EXISTS follows (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  follower_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  following_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  created_at timestamptz DEFAULT now(),
  UNIQUE(follower_id, following_id)
);
ALTER TABLE follows ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "read follows" ON follows;
CREATE POLICY "read follows" ON follows FOR SELECT USING (true);
DROP POLICY IF EXISTS "own follows" ON follows;
CREATE POLICY "own follows" ON follows FOR ALL
  USING (auth.uid() = follower_id)
  WITH CHECK (auth.uid() = follower_id);

-- Enable fuzzy search for username lookup
CREATE EXTENSION IF NOT EXISTS pg_trgm;
CREATE INDEX IF NOT EXISTS profiles_username_trgm
  ON profiles USING gin(username gin_trgm_ops);

// File: src/app/auth/page.js
'use client'
import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'
import { useRouter } from 'next/navigation'

function validatePassword(pw) {
  const errors = []
  if (pw.length < 8) errors.push('At least 8 characters')
  if (!/[A-Z]/.test(pw)) errors.push('One uppercase letter (A–Z)')
  if (!/[0-9]/.test(pw)) errors.push('One number (0–9)')
  if (!/[!@#$%^&*()_+\-=\[\]{};':"|,.<>\/?]/.test(pw))
    errors.push('One special character (! @ # $ ...)')
  return errors
}

export default function Auth() {
  const router = useRouter()
  const [mode, setMode] = useState('signin')
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [username, setUsername] = useState('')
  const [message, setMessage] = useState('')
  const [loading, setLoading] = useState(false)

  useEffect(() => {
    supabase.auth.getUser().then(({ data: { user } }) => {
      if (user) router.push('/profile')
    })
  }, [])

  const signIn = async () => {
    if (!email || !password) { setMessage('Please enter email and password.'); return }
    setLoading(true)
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) { setMessage(error.message); setLoading(false) }
    else { router.push('/profile') }
  }

  const signUp = async () => {
    if (!email || !password) { setMessage('Please enter email and password.'); return }
    const pwErrors = validatePassword(password)
    if (pwErrors.length > 0) {
      setMessage('Password needs: ' + pwErrors.join(' · '))
      return
    }
    setLoading(true)
    const { data, error } = await supabase.auth.signUp({ email, password })
    if (error) { setMessage(error.message); setLoading(false); return }
    if (data?.user) {
      await supabase.from('profiles').upsert({
        id: data.user.id,
        username: username || email.split('@')[0],
        email,
      }, { onConflict: 'id', ignoreDuplicates: true })
    }
    setMessage('Account created! Check your email to confirm, then sign in.')
    setLoading(false)
  }

  const onKey = (e) => { if (e.key === 'Enter') mode === 'signin' ? signIn() : signUp() }
  const pwErrors = mode === 'signup' && password.length > 0 ? validatePassword(password) : []

  return (
    <main className="section" style={{ paddingTop: '5rem' }}>
      <div style={{ maxWidth: 440, margin: '0 auto' }}>
        <div style={{ textAlign: 'center', marginBottom: '2.5rem' }}>
          <h1 style={{ fontSize: '2.2rem', fontWeight: 700, marginBottom: '0.5rem' }}>
            {mode === 'signin' ? 'Welcome back' : 'Join Banger Ratios'}
          </h1>
          <p style={{ color: 'var(--gray-text)' }}>
            {mode === 'signin' ? 'Sign in to rate albums.' : 'Create an account to start rating.'}
          </p>
        </div>
        <div style={{ background: 'white', padding: '2.5rem', borderRadius: '20px',
          border: '1px solid var(--border)', boxShadow: '0 4px 24px rgba(0,0,0,0.05)' }}>
          <div style={{ display: 'flex', marginBottom: '2rem',
            border: '1px solid var(--border)', borderRadius: '10px', overflow: 'hidden' }}>
            {['signin','signup'].map(m => (
              <button key={m} onClick={() => { setMode(m); setMessage('') }} style={{
                flex: 1, padding: '0.6rem', border: 'none', cursor: 'pointer',
                background: mode === m ? 'var(--pink)' : 'white',
                color: mode === m ? 'white' : 'var(--gray-text)',
                fontWeight: 600, fontSize: '0.85rem', fontFamily: 'inherit',
              }}>{m === 'signin' ? 'Sign In' : 'Create Account'}</button>
            ))}
          </div>
          {mode === 'signup' && (
            <div style={{ marginBottom: '1rem' }}>
              <label style={{ display: 'block', fontWeight: 600, fontSize: '0.85rem', marginBottom: '0.4rem' }}>Username</label>
              <input className="input" type="text" placeholder="your_username"
                value={username} onChange={e => setUsername(e.target.value)} onKeyDown={onKey} />
            </div>
          )}
          <div style={{ marginBottom: '1rem' }}>
            <label style={{ display: 'block', fontWeight: 600, fontSize: '0.85rem', marginBottom: '0.4rem' }}>Email</label>
            <input className="input" type="email" placeholder="you@example.com"
              value={email} onChange={e => setEmail(e.target.value)} onKeyDown={onKey} />
          </div>
          <div style={{ marginBottom: '0.5rem' }}>
            <label style={{ display: 'block', fontWeight: 600, fontSize: '0.85rem', marginBottom: '0.4rem' }}>Password</label>
            <input className="input" type="password"
              placeholder={mode === 'signup' ? 'Min 8 chars, uppercase, number, symbol' : '••••••••'}
              value={password} onChange={e => setPassword(e.target.value)} onKeyDown={onKey} />
          </div>
          {pwErrors.length > 0 && (
            <div style={{ marginBottom: '1rem' }}>
              {pwErrors.map((err,i) => (
                <p key={i} style={{ color: '#dc2626', fontSize: '0.78rem', marginTop: '0.2rem' }}>✗ {err}</p>
              ))}
            </div>
          )}
          {pwErrors.length === 0 && mode === 'signup' && password.length >= 8 && (
            <p style={{ color: '#00B84D', fontSize: '0.78rem', marginBottom: '1rem' }}>✓ Password looks good</p>
          )}
          <div style={{ marginBottom: '1.5rem' }} />
          <button onClick={mode === 'signin' ? signIn : signUp} disabled={loading}
            className="primary-btn" style={{ width: '100%', textAlign: 'center', opacity: loading ? 0.7 : 1 }}>
            {loading ? 'Please wait...' : mode === 'signin' ? 'Sign In' : 'Create Account'}
          </button>
          {message && (
            <div style={{ marginTop: '1rem', padding: '0.75rem 1rem', borderRadius: '8px',
              background: message.includes('created') ? '#f0fff4' : '#fff0f5',
              color: message.includes('created') ? '#00B84D' : '#cc0044',
              fontSize: '0.85rem', fontWeight: 500 }}>
              {message}
            </div>
          )}
        </div>
        <p style={{ textAlign: 'center', marginTop: '1.5rem', color: 'var(--gray-text)', fontSize: '0.82rem' }}>
          By signing in, you agree to help us expose mid albums.
        </p>
      </div>
    </main>
  )
}
EOF

// File: src/app/components/Nav.js
'use client'
import Link from 'next/link'
import { usePathname, useRouter } from 'next/navigation'
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'

export default function Nav() {
  const pathname = usePathname()
  const router = useRouter()
  const [user, setUser] = useState(null)
  const [menuOpen, setMenuOpen] = useState(false)

  useEffect(() => {
    supabase.auth.getUser().then(({ data }) => setUser(data?.user || null))
    const { data: { subscription } } = supabase.auth.onAuthStateChange((_e, session) => {
      setUser(session?.user || null)
    })
    return () => subscription.unsubscribe()
  }, [])

  const signOut = async () => {
    await supabase.auth.signOut()
    setUser(null)
    router.push('/')
  }

  const links = [
    { href: '/', label: 'HOME' },
    { href: '/albums', label: 'ALBUMS' },
    { href: '/artists', label: 'ARTISTS' },
    { href: '/leaderboards', label: 'LEADERBOARD' },
    { href: '/releases', label: 'RELEASES' },
    { href: '/friends', label: 'FRIENDS' },
    { href: '/about', label: 'ABOUT' },
    { href: '/merch', label: 'MERCH' },
  ]

  return (
    <nav className="nav">
      <div className="nav-inner">
        <Link href="/" className="logo">
          <span className="logo-square" />
          BANGER <span className="logo-pink">RATIOS</span>
        </Link>
        <div className="nav-right">
          {links.map(({ href, label }) => (
            <Link key={href} href={href} className="nav-link nav-link-desktop">
              <span className={pathname === href ? 'active' : ''}>{label}</span>
            </Link>
          ))}
          {user ? (
            <>
              <Link href="/profile" className="nav-link nav-link-desktop">
                <span className={pathname === '/profile' ? 'active' : ''}>PROFILE</span>
              </Link>
              <button onClick={signOut} className="nav-link-desktop" style={{
                background: 'none', border: 'none', cursor: 'pointer', padding: 0,
                fontFamily: 'inherit', fontWeight: 500, fontSize: '0.82rem',
                letterSpacing: '1.5px', color: 'var(--gray-text)',

            </>

                padding: '0.45rem 1.2rem', borderRadius: '8px',
                fontSize: '0.78rem', fontWeight: 700, letterSpacing: '1px',

      if (!user) { router.push('/auth'); return }
      setMe(user)
      const { data: follows } = await supabase.from('follows')
        .select('following_id').eq('follower_id', user.id)
      const ids = (follows || []).map(f => f.following_id)
      setFollowing(new Set(ids))
      if (ids.length > 0) {
        const { data: activity } = await supabase.from('ratings')
          .select(`score, created_at,
            profiles ( username ),
            tracks ( name ),

async function uploadFile(file, bucket, suffix) {
  const ext = file.name.split('.').pop()
  const path = `${user.id}/${suffix}.${ext}`
  const { error } = await supabase.storage.from(bucket).upload(path, file, { upsert: true })
  if (error) { console.error(error); return null }
  const { data: { publicUrl } } = supabase.storage.from(bucket).getPublicUrl(path)
  return publicUrl
}

async function uploadBanner(file) {
  const url = await uploadFile(file, 'banners', 'banner')
  if (!url) return
  await supabase.from('profiles').update({ banner_url: url }).eq('id', user.id)
  setProfile(prev => ({ ...prev, banner_url: url }))
}

async function uploadAvatar(file) {
  const url = await uploadFile(file, 'avatars', 'avatar')
  if (!url) return
  await supabase.from('profiles').update({ avatar_url: url }).eq('id', user.id)
  setProfile(prev => ({ ...prev, avatar_url: url }))
}

async function connectSpotify() {
  await supabase.auth.signInWithOAuth({
    provider: 'spotify',
    options: {
      scopes: 'user-read-recently-played user-top-read user-read-private',
      redirectTo: `${window.location.origin}/profile`,
    }
  })
}
