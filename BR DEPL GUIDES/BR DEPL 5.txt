BLOCK 6 — Create the recalculate_banger_ratio function
This function runs every time someone submits ratings. It recalculates whether each track is a Banger, then updates the album's Banger Ratio percentage. Without this, ratings don't update anything.
  SQL → paste in Supabase SQL Editor → click Run
-- BLOCK 6: The recalculate_banger_ratio function
-- This runs every time someone submits ratings.
-- Safe to re-run — replaces existing version.
 
CREATE OR REPLACE FUNCTION recalculate_banger_ratio(p_album_id bigint)
RETURNS void LANGUAGE plpgsql AS $$
BEGIN
  -- Step 1: Update each track with its aggregate stats
  UPDATE tracks t SET
    avg_rating = (
      SELECT AVG(score)::numeric(4,2)
      FROM ratings r WHERE r.track_id = t.id
    ),
    total_ratings = (
      SELECT COUNT(*) FROM ratings r WHERE r.track_id = t.id
    ),
    is_banger = (
      SELECT AVG(score) >= 5 FROM ratings r WHERE r.track_id = t.id
    )
  WHERE t.album_id = p_album_id;
 
  -- Step 2: Update the album banger_ratio
  UPDATE albums SET
    banger_ratio = (
      SELECT ROUND(
        100.0 * COUNT(*) FILTER (WHERE is_banger) / NULLIF(COUNT(*),0),
      2)
      FROM tracks WHERE album_id = p_album_id AND total_ratings > 0
    ),
    total_ratings = (
      SELECT COUNT(*) FROM ratings WHERE album_id = p_album_id
    ),
    updated_at = now()
  WHERE id = p_album_id;
END;
$$;
 
BLOCK 7 — Set up Row Level Security
This controls who can read/write what. Public can read everything (for the leaderboard). Only logged-in users can write their own ratings and profile.
  SQL → paste in Supabase SQL Editor → click Run
-- BLOCK 7: Row Level Security (RLS)
-- Controls who can read and write what.
-- Run this entire block at once.
 
ALTER TABLE albums  ENABLE ROW LEVEL SECURITY;
ALTER TABLE tracks  ENABLE ROW LEVEL SECURITY;
ALTER TABLE ratings ENABLE ROW LEVEL SECURITY;
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
 
-- ANYONE can read albums, tracks, ratings (needed for leaderboard)
DROP POLICY IF EXISTS "read albums" ON albums;
CREATE POLICY "read albums" ON albums FOR SELECT USING (true);
 
DROP POLICY IF EXISTS "read tracks" ON tracks;
CREATE POLICY "read tracks" ON tracks FOR SELECT USING (true);
 
DROP POLICY IF EXISTS "read ratings" ON ratings;
CREATE POLICY "read ratings" ON ratings FOR SELECT USING (true);
 
DROP POLICY IF EXISTS "read profiles" ON profiles;
CREATE POLICY "read profiles" ON profiles FOR SELECT USING (true);
 
-- LOGGED-IN users can write albums and tracks (when they rate an album)
DROP POLICY IF EXISTS "write albums" ON albums;
CREATE POLICY "write albums" ON albums FOR ALL
  USING (auth.role() = 'authenticated')
  WITH CHECK (auth.role() = 'authenticated');
 
DROP POLICY IF EXISTS "write tracks" ON tracks;
CREATE POLICY "write tracks" ON tracks FOR ALL
  USING (auth.role() = 'authenticated')
  WITH CHECK (auth.role() = 'authenticated');
 
-- Users can only write THEIR OWN ratings
DROP POLICY IF EXISTS "own ratings" ON ratings;
CREATE POLICY "own ratings" ON ratings FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);
 
-- Users can only write THEIR OWN profile
DROP POLICY IF EXISTS "own profile" ON profiles;
CREATE POLICY "own profile" ON profiles FOR ALL
  USING (auth.uid() = id)
  WITH CHECK (auth.uid() = id);
 
BLOCK 8 — Auto-create profile on signup
THIS IS THE KEY FIX for the "profile already exists" error and the blank profile page. This creates a trigger — think of it like a rule that fires automatically. Whenever a new user signs up, Supabase automatically creates a row in your profiles table for them. No more manual steps, no more errors.
  SQL → paste in Supabase SQL Editor → click Run
-- BLOCK 8: Auto-create profile when user signs up
-- This FIXES the "profile already exists" error you were seeing.
-- Now a profile row is created automatically on every signup.
 
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  INSERT INTO profiles (id, email, username, created_at)
  VALUES (
    new.id,
    new.email,
    split_part(new.email, '@', 1),
    now()
  )
  ON CONFLICT (id) DO NOTHING;  -- prevents "already exists" error
  RETURN new;
END;
$$;
 
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

function AnimatedCount({ value }) {
  const [display, setDisplay] = useState(0)
  useEffect(() => {
    if (value === null) return
    const steps = 50
    const increment = value / steps
    let step = 0
    const timer = setInterval(() => {
      step++
      setDisplay(Math.min(Math.round(increment * step), value))
      if (step >= steps) clearInterval(timer)
    }, 1200 / steps)
    return () => clearInterval(timer)
  }, [value])
  if (value === null) return <span style={{ color: "#ddd" }}>—</span>
  const n = display
  const f = n >= 1000000 ? (n/1000000).toFixed(1)+'M' : n >= 1000 ? (n/1000).toFixed(1)+'K' : n.toLocaleString()
  return <span>{f}</span>
}

import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'
import { useRouter } from 'next/navigation'
 
export default function Auth() {
  const router = useRouter()
  const [mode, setMode] = useState('signin')
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [username, setUsername] = useState('')
  const [message, setMessage] = useState('')
  const [loading, setLoading] = useState(false)
 
  useEffect(() => {
    supabase.auth.getUser().then(({ data: { user } }) => { if (user) router.push('/profile') })
  }, [])
 
  const signIn = async () => {
    if (!email || !password) { setMessage('Please enter email and password.'); return }
    setLoading(true)
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) { setMessage(error.message); setLoading(false) } else { router.push('/profile') }
  }
 
  const signUp = async () => {
    if (!email || !password) { setMessage('Please enter email and password.'); return }
    if (password.length < 6) { setMessage('Password must be at least 6 characters.'); return }
    setLoading(true)
    const { data, error } = await supabase.auth.signUp({ email, password })
    if (error) { setMessage(error.message); setLoading(false); return }
    if (data?.user) {
      await supabase.from('profiles').upsert({
        id: data.user.id, username: username || email.split('@')[0], email,
      }, { onConflict: 'id', ignoreDuplicates: true })
    }
    setMessage('Account created! Check your email to confirm, then sign in.')
    setLoading(false)
  }
 
  const key = (e) => { if (e.key === "Enter") mode === "signin" ? signIn() : signUp() }
 
  return (

            {mode === "signin" ? "Sign in to rate albums and track your scores." : "Create an account to start rating."}

        </div>
        <div style={{ background: "white", padding: "2.5rem", borderRadius: "20px",
          border: "1px solid var(--border)", boxShadow: "0 4px 24px rgba(0,0,0,0.05)" }}>

            border: "1px solid var(--border)", borderRadius: "10px", overflow: "hidden" }}>

              <button key={m} onClick={() => { setMode(m); setMessage("") }} style={{
                flex: 1, padding: "0.6rem", border: "none", cursor: "pointer",

                color: mode === m ? "white" : "var(--gray-text)",

              }}>{m === "signin" ? "Sign In" : "Create Account"}</button>
            ))}

          {mode === "signup" && (
            <div style={{ marginBottom: "1rem" }}>
              <label style={{ display: "block", fontWeight: 600, fontSize: "0.85rem", marginBottom: "0.4rem" }}>Username</label>

'use client'
import { useEffect, useState } from 'react'
import Link from 'next/link'
 
const GENRE_IDS = {
  'All': '',
  'Hip-Hop': '18',
  'R&B': '15',
  'Pop': '14',
  'Rock': '21',
  'Electronic': '7',
  'Alternative': '20',
}
 
export default function Releases() {
  const [albums, setAlbums] = useState([])
  const [genre, setGenre] = useState('All')
  const [loading, setLoading] = useState(true)
 
  useEffect(() => {
    fetchReleases(genre)
  }, [genre])
 
  async function fetchReleases(selectedGenre) {
    setLoading(true)
    try {
      const genreId = GENRE_IDS[selectedGenre]
      // iTunes RSS feed for new releases — no API key required
      const url = genreId
        ? `https://itunes.apple.com/us/rss/topalbums/limit=24/genre=${genreId}/json`
        : `https://itunes.apple.com/us/rss/topalbums/limit=24/json`
 
      const res = await fetch(url)
      const data = await res.json()
      const entries = data?.feed?.entry || []
 
      const parsed = entries.map(e => ({
        id: e.id?.attributes?.['im:id'],
        title: e['im:name']?.label,
        artist: e['im:artist']?.label,
        artwork: e['im:image']?.[2]?.label?.replace('170x170', '300x300'),
        genre: e.category?.attributes?.label,
        price: e['im:price']?.label,
        releaseDate: e['im:releaseDate']?.label,
      }))
 
      setAlbums(parsed.filter(a => a.id && a.title))
    } catch {
      setAlbums([])
    }
    setLoading(false)
  }
 
  return (
    <main className="section">
      <div className="page-header">
        <h1>New Releases</h1>
        <p>Top albums right now — click any to rate it and calculate the Banger Ratio.</p>

            style={{

        </div>

          display: 'grid',

          {albums.map(album => (

              <div className="card" style={{ padding: '0', overflow: 'hidden', cursor: 'pointer' }}>
                <div style={{ position: 'relative' }}>
                  {album.artwork ? (
                    <img src={album.artwork} alt={album.title}
                      style={{ width: '100%', aspectRatio: '1', objectFit: 'cover', display: 'block' }} />

                    overflow: 'hidden', textOverflow: 'ellipsis', whiteSpace: 'nowrap' }}>
                    {album.title}

                    overflow: 'hidden', textOverflow: 'ellipsis', whiteSpace: 'nowrap' }}>
                    {album.artist}
                  </p>
                  {album.genre && (
                    <p style={{ fontSize: '0.7rem', color: 'var(--gray-text)', marginTop: '0.4rem' }}>
                      {album.genre}
                    </p>
                  )}
                </div>
              </div>
            </Link>
          ))}
        </div>
      )}
 
      {!loading && albums.length === 0 && (
        <div style={{ textAlign: 'center', padding: '4rem', color: 'var(--gray-text)' }}>
          No releases found for this genre right now.
        </div>
      )}
    </main>
  )
}
