-- =============================================
-- BANGER RATIOS BASE SCHEMA (V1)
-- =============================================

-- 1. PROFILES TABLE
create table public.profiles (
  id uuid references auth.users on delete cascade primary key,
  username text unique,
  display_name text,
  avatar_url text,
  is_premium boolean default false,
  created_at timestamptz default now()
);

-- 2. ALBUMS TABLE
create table public.albums (
  id bigserial primary key,
  itunes_collection_id bigint unique not null,
  name text not null,
  artist_name text not null,
  artwork_url text,
  release_date date,
  genre text,
  track_count integer default 0,
  total_ratings integer default 0,
  banger_ratio numeric(5,2) default 0,
  classification text default 'Unrated',
  created_at timestamptz default now()
);

-- 3. TRACKS TABLE
create table public.tracks (
  id bigserial primary key,
  album_id bigint references public.albums on delete cascade,
  itunes_track_id bigint unique not null,
  name text not null,
  track_number integer not null,
  duration_ms integer,
  preview_url text,
  avg_rating numeric(3,1) default 0,
  total_ratings integer default 0,
  is_banger boolean default false,
  created_at timestamptz default now()
);

-- 4. RATINGS TABLE (1-7 scale)
create table public.ratings (
  id bigserial primary key,
  user_id uuid references public.profiles on delete cascade not null,
  track_id bigint references public.tracks on delete cascade not null,
  album_id bigint references public.albums on delete cascade not null,
  score smallint not null check (score >= 1 and score <= 7),
  created_at timestamptz default now(),
  unique(user_id, track_id)
);

-- 5. ROW LEVEL SECURITY
alter table public.profiles enable row level security;
alter table public.albums enable row level security;
alter table public.tracks enable row level security;
alter table public.ratings enable row level security;

-- 6. SECURITY POLICIES
create policy "Public read albums" on public.albums for select using (true);
create policy "Public read tracks" on public.tracks for select using (true);
create policy "Public read profiles" on public.profiles for select using (true);
create policy "Public read ratings" on public.ratings for select using (true);
create policy "Users insert ratings" on public.ratings
  for insert with check (auth.uid() = user_id);
create policy "Users update own ratings" on public.ratings
  for update using (auth.uid() = user_id);
create policy "Users delete own ratings" on public.ratings
  for delete using (auth.uid() = user_id);
create policy "Users update own profile" on public.profiles
  for update using (auth.uid() = id);
create policy "Users insert own profile" on public.profiles
  for insert with check (auth.uid() = id);
create policy "Auth users insert albums" on public.albums
  for insert with check (auth.uid() is not null);
create policy "Auth users update albums" on public.albums
  for update using (auth.uid() is not null);
create policy "Auth users insert tracks" on public.tracks
  for insert with check (auth.uid() is not null);

-- 7. AUTO-CREATE PROFILE ON SIGNUP
create or replace function public.handle_new_user()
returns trigger as $$
begin
  insert into public.profiles (id, username, display_name)
  values (new.id, new.raw_user_meta_data->>'username', new.raw_user_meta_data->>'display_name');
  return new;
end;
$$ language plpgsql security definer;

create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();

-- 8. RECALCULATE BANGER RATIO FUNCTION (banger = avg >= 5.0)
create or replace function public.recalculate_banger_ratio(p_album_id bigint)
returns void as $$
declare
  v_total_tracks integer; v_banger_tracks integer;
  v_ratio numeric(5,2); v_class text;
begin
  update public.tracks t set
    avg_rating = sub.avg_score, total_ratings = sub.cnt, is_banger = (sub.avg_score >= 5.0)
  from (select track_id, round(avg(score)::numeric,1) as avg_score, count(*) as cnt
        from public.ratings where album_id = p_album_id group by track_id) sub
  where t.id = sub.track_id and t.album_id = p_album_id;

  select count(*) into v_total_tracks from public.tracks where album_id = p_album_id and total_ratings > 0;
  select count(*) into v_banger_tracks from public.tracks where album_id = p_album_id and is_banger = true;

  if v_total_tracks > 0 then
    v_ratio := round((v_banger_tracks::numeric / v_total_tracks) * 100, 2);
  else v_ratio := 0; end if;

  v_class := case when v_ratio >= 90 then 'Certified Classic'
                  when v_ratio >= 75 then 'Solid Gold'
                  when v_ratio >= 60 then 'Hit or Miss'
                  when v_ratio >= 40 then 'Filler Warning'
                  else 'Skip It' end;

  update public.albums set banger_ratio = v_ratio, classification = v_class,
    total_ratings = (select count(*) from public.ratings where album_id = p_album_id)
  where id = p_album_id;
end;
$$ language plpgsql security definer;

                <label style={{ fontSize: 11, color: '#999', fontWeight: 600, display: 'block', marginBottom: 4 }}>{label}</label>
                {isTA
                  ? <textarea style={{...inp, minHeight:60, resize:"vertical"}} value={form[key]} onChange={e => setForm({...form, [key]: e.target.value})} placeholder={ph} />
                  : <input style={inp} value={form[key]} onChange={e => setForm({...form, [key]: e.target.value})} placeholder={ph} />}
              </div>
            ))}
            <div style={{ display: 'flex', gap: 8 }}>
              <button onClick={saveProfile} style={{ padding: '9px 22px', borderRadius: 8, border: 'none',
                background: '#FF0066', color: 'white', fontSize: 13, fontWeight: 700, cursor: 'pointer' }}>
                {saving ? 'Saving...' : 'Save'}</button>
              <button onClick={() => setEditing(false)} style={{ padding: '9px 22px', borderRadius: 8,
                border: '1px solid #E5E5E5', background: 'white', color: '#666', fontSize: 13, cursor: 'pointer' }}>
                Cancel</button>
            </div>
          </div>
        )}
      </div>

      {/* STATS */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 1fr)', gap: 10, marginBottom: 24 }}>
        {[
          { label: 'Albums Rated', value: myRatings.length },
          { label: 'Total Ratings', value: myRatings.reduce((s,a) => s + (a.total_ratings||0), 0) },
          { label: 'Avg Ratio', value: myRatings.length ? (myRatings.reduce((s,a) => s + parseFloat(a.banger_ratio||0), 0)/myRatings.length).toFixed(0)+'%' : '--' },
        ].map((s,i) => (
          <div key={i} style={{ background: '#F9F9F9', borderRadius: 12, padding: 16, textAlign: 'center' }}>
            <p style={{ fontSize: 24, fontWeight: 700, color: '#FF0066' }}>{s.value}</p>
            <p style={{ fontSize: 11, color: '#999', fontWeight: 500 }}>{s.label}</p>
          </div>
        ))}
      </div>

      <h2 style={{ fontSize: 18, fontWeight: 700, marginBottom: 12 }}>Albums I’ve Rated</h2>
      {myRatings.length === 0 && <p style={{ color: '#CCC', fontSize: 14 }}>No albums rated yet.{' '}
        <a href="/" style={{ color: "#FF0066" }}>Start rating!</a></p>}
      {myRatings.map(a => (
        <a key={a.id} href={`/album/${a.itunes_collection_id}`}
          style={{ display: 'flex', alignItems: 'center', gap: 12, padding: '10px 12px',
            borderRadius: 10, marginBottom: 2, transition: 'background 0.15s' }}
          onMouseOver={e => e.currentTarget.style.background = '#F5F5F5'}
          onMouseOut={e => e.currentTarget.style.background = ''}>
          {a.artwork_url && <img src={a.artwork_url.replace("600x600","80x80")} alt=""
            style={{ width: 40, height: 40, borderRadius: 6 }} />}
          <div style={{ flex: 1 }}>
            <p style={{ fontSize: 13, fontWeight: 600 }}>{a.name}</p>
            <p style={{ fontSize: 11, color: '#999' }}>{a.artist_name}</p>
          </div>
          <span style={{ fontSize: 16, fontWeight: 700, color: '#FF0066' }}>{a.banger_ratio}%</span>
        </a>
      ))}
    </div>
  )
}

Troubleshooting

Problem	Fix
Vercel: "Can't resolve '@/lib/supabase'"	Do Steps 2.5 and 2.6. Create jsconfig.json and fix next.config.mjs. Commit and push again.
Vercel: blank white page after deploy	Missing environment variables. Go to Project Settings → Environment Variables, add both keys, Redeploy.
Local dev: blank page or errors	Check .env.local has correct values. No spaces around the = sign.
Old .tsx files conflicting	If you have both page.tsx and page.js in the same folder, delete the .tsx file.
Album art not loading on Vercel	You need the next.config.mjs fix from Step 2.5 with the iTunes domains whitelisted.
Ratings not saving	User must be signed in AND email-confirmed. Check Supabase RLS policies are active.
git push asks for a password	Use a Personal Access Token (github.com → Settings → Developer settings → Personal access tokens), not your GitHub password.
Search shows no results	iTunes API occasionally rate-limits. Wait 30 seconds and try again.
