PHASE 0 — UNDERSTAND YOUR STACK (5 min read)
Before touching any code, here is exactly what each tool does and why you need it. You only need to read this once.

TERMINAL	A text-based way to control your computer. Instead of clicking folders you type commands. On Mac: Cmd+Space, type Terminal, press Enter.
VS CODE	A code editor. Think of it as Word for code. You mostly will NOT need this — the Python Autopatcher writes files for you.
GITHUB	Online storage for your code. Think Google Drive, but for code. Vercel reads from GitHub to deploy your site.
SUPABASE	Your database. Stores user accounts, album ratings, leaderboard data, profile photos, and all relationships between them.
VERCEL	Puts your website on the internet for free. It watches GitHub — every time you push code, it rebuilds your site automatically.
NEXT.JS	The framework your website is built in. React + server-side rendering. You do not need to understand it deeply — the scripts handle it.

-- PROFILES (one row per user, created automatically on signup)
create table if not exists public.profiles (
  id uuid references auth.users(id) on delete cascade primary key,
  username text unique,
  display_name text,
  bio text,
  avatar_url text,
  apple_music_url text,
  spotify_url text,
  favorite_albums jsonb default '[]'::jsonb,
  created_at timestamptz default now()
);

-- ALBUMS
create table if not exists public.albums (
  id bigserial primary key,
  title text not null,
  artist text not null,
  genre text,
  year int,
  itunes_id text,
  artwork_url text,
  created_at timestamptz default now()
);

-- TRACKS
create table if not exists public.tracks (
  id bigserial primary key,
  album_id bigint references public.albums(id) on delete cascade,
  title text not null,
  track_number int,
  created_at timestamptz default now()
);

-- RATINGS (one row per user per track)
create table if not exists public.ratings (
  id bigserial primary key,
  user_id uuid references auth.users(id) on delete cascade,
  track_id bigint references public.tracks(id) on delete cascade,
  album_id bigint references public.albums(id) on delete cascade,
  is_banger boolean not null,
  created_at timestamptz default now(),
  unique(user_id, track_id)
);

-- FOLLOWS (friend connections)
create table if not exists public.follows (
  follower_id uuid references auth.users(id) on delete cascade,
  following_id uuid references auth.users(id) on delete cascade,
  created_at timestamptz default now(),
  primary key (follower_id, following_id)
);

-- LISTS (custom user lists)
create table if not exists public.lists (
  id bigserial primary key,
  user_id uuid references auth.users(id) on delete cascade,
  name text not null,
  created_at timestamptz default now()
);

-- LIST ITEMS
create table if not exists public.list_items (
  id bigserial primary key,
  list_id bigint references public.lists(id) on delete cascade,
  album_id bigint references public.albums(id) on delete cascade,
  created_at timestamptz default now()
);

-- ── ROW LEVEL SECURITY ─────────────────────────────────────
alter table public.profiles enable row level security;
alter table public.albums enable row level security;
alter table public.tracks enable row level security;
alter table public.ratings enable row level security;
alter table public.follows enable row level security;
alter table public.lists enable row level security;
alter table public.list_items enable row level security;

-- PROFILES: anyone can read, only owner can write
create policy "profiles_read_all" on public.profiles for select using (true);
create policy "profiles_insert_own" on public.profiles for insert with check (auth.uid() = id);
create policy "profiles_update_own" on public.profiles for update using (auth.uid() = id);

-- ALBUMS: anyone can read, authenticated users can insert
create policy "albums_read_all" on public.albums for select using (true);
create policy "albums_insert_auth" on public.albums for insert with check (auth.role() = 'authenticated');

-- TRACKS: anyone can read
create policy "tracks_read_all" on public.tracks for select using (true);
create policy "tracks_insert_auth" on public.tracks for insert with check (auth.role() = 'authenticated');

-- RATINGS: anyone can read, only owner can write
create policy "ratings_read_all" on public.ratings for select using (true);
create policy "ratings_insert_own" on public.ratings for insert with check (auth.uid() = user_id);
create policy "ratings_update_own" on public.ratings for update using (auth.uid() = user_id);
create policy "ratings_delete_own" on public.ratings for delete using (auth.uid() = user_id);

-- FOLLOWS: anyone can read, owner can insert/delete
create policy "follows_read_all" on public.follows for select using (true);
create policy "follows_insert_own" on public.follows for insert with check (auth.uid() = follower_id);
create policy "follows_delete_own" on public.follows for delete using (auth.uid() = follower_id);

-- LISTS and LIST_ITEMS: owner only
create policy "lists_read_own" on public.lists for select using (auth.uid() = user_id);
create policy "lists_insert_own" on public.lists for insert with check (auth.uid() = user_id);
create policy "list_items_read_own" on public.list_items for select using (
  exists (select 1 from public.lists where id = list_id and user_id = auth.uid())
);
create policy "list_items_insert_own" on public.list_items for insert with check (
  exists (select 1 from public.lists where id = list_id and user_id = auth.uid())
);

-- ── STORAGE BUCKET FOR AVATARS ──────────────────────────────
insert into storage.buckets (id, name, public)
values ('avatars', 'avatars', true)
on conflict (id) do nothing;

create policy "avatars_read_all" on storage.objects for select using (bucket_id = 'avatars');
create policy "avatars_upload_own" on storage.objects for insert with check (
  bucket_id = 'avatars' and auth.uid()::text = (storage.foldername(name))[1]
);
create policy "avatars_update_own" on storage.objects for update using (
  bucket_id = 'avatars' and auth.uid()::text = (storage.foldername(name))[1]
);

-- ── AUTO-CREATE PROFILE ON SIGNUP ───────────────────────────
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer set search_path = public
as $$
begin
  insert into public.profiles (id, username, display_name)
  values (
    new.id,
    split_part(new.email, '@', 1),
    split_part(new.email, '@', 1)
  );
  return new;
end;
$$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();

-- ── BANGER RATIO CALCULATION ─────────────────────────────────
create or replace function public.recalculate_banger_ratio(p_album_id bigint)
returns numeric language plpgsql security definer
as $$
declare
  total_ratings int;
  banger_ratings int;
begin
  select count(*) into total_ratings
    from public.ratings where album_id = p_album_id;
  select count(*) into banger_ratings
    from public.ratings where album_id = p_album_id and is_banger = true;
  if total_ratings = 0 then return 0; end if;
  return round((banger_ratings::numeric / total_ratings::numeric) * 100, 1);
end;
$$;

PHASE 4 — THE PYTHON AUTOPATCHER (THE BIG ONE)
This is the most important section. A single Python script writes ALL of your page files with correct, professional code. Run it once and every page is fixed.

WHY PYTHON INSTEAD OF HEREDOC
The heredoc (cat << EOF) method caused the NewsBanner.js corruption in DEP XIV — terminal commands ended up inside your JavaScript files.
Python writes files directly without any shell interpretation. It cannot corrupt your files.
This script is self-contained. Copy the whole thing, paste it into Terminal, press Enter.

Step 4.2 — Run the Diagnostic Script first
This scans your project and flags any files with the broken "missing import" pattern. Run this before the Autopatcher so you can see exactly what is broken.
🖥️  YOU ARE IN: TERMINAL
Copy and paste the command(s) below exactly.
python3 << 'PYEOF'
import os, glob
BASE = os.path.expanduser("~/Desktop/banger-ratios/src/app")
pages = glob.glob(BASE + "/**/page.js", recursive=True)
print(f"Found {len(pages)} page files:")
for p in sorted(pages):
    short = p.replace(os.path.expanduser("~"), "~")
    with open(p) as f: content = f.read()
    has_import = "import { useEffect" in content or "import {useEffect" in content
    has_bare_use = "useEffect(" in content and not has_import
    status = "BROKEN (missing import)" if has_bare_use else "ok"
    print(f"  [{status}]  {short}")
PYEOF

import { useEffect, useState } from 'react'

  const [saving, setSaving] = useState(false)
  const [message, setMessage] = useState("")

  useEffect(() => {
    supabase.auth.getUser().then(({ data }) => {
      if (!data.user) { window.location.href = "/profile"; return }
      setUser(data.user)
      supabase.from("profiles").select("*").eq("id", data.user.id).single()
        .then(({ data: p }) => {
          setProfile(p || {})
          if (p?.favorite_albums?.length) {
            const arr = [...p.favorite_albums]
            while (arr.length < 5) arr.push("")
            setFavAlbums(arr)
          }
        })
    })

      }

        id: user.id,
        username: profile.username,
        display_name: profile.display_name,
        bio: profile.bio,
        avatar_url,
        apple_music_url: profile.apple_music_url,
        spotify_url: profile.spotify_url,
        favorite_albums: favAlbums.filter(Boolean),
      })
      if (newPassword) {
        await supabase.auth.updateUser({ password: newPassword })
      }
      setMessage("Saved!")
    } catch (e) {
      setMessage("Error saving: " + e.message)
    }
    setSaving(false)
  }

      <label style={{ display:"block", fontWeight:600, marginBottom:6, fontSize:"0.85rem" }}>{label}</label>

        style={{ width:"100%", padding:"10px 14px", borderRadius:8, border:"1px solid #ddd",
          fontFamily:"inherit", fontSize:"0.9rem" }} />
    </div>

  return (

      <h1 style={{ fontSize: "2rem", fontWeight: 800, marginBottom: 32 }}>Settings</h1>
      {field("Display Name", "display_name", "text", "Your public name")}
      {field("Username", "username", "text", "@username")}
      {field("Bio", "bio", "text", "A short bio...")}
      {field("Apple Music URL", "apple_music_url", "url", "https://music.apple.com/...")}
      {field("Spotify URL", "spotify_url", "url", "https://open.spotify.com/...")}
      <div style={{ marginBottom: 16 }}>

// File: src/middleware.js
import { NextResponse } from 'next/server'

const RATE_LIMIT = 60
const WINDOW_MS = 60 * 1000
const store = new Map()

export function middleware(req) {
  const ip = req.headers.get("x-forwarded-for") || "unknown"
  const now = Date.now()
  const entry = store.get(ip) || { count: 0, start: now }
  if (now - entry.start > WINDOW_MS) {
    store.set(ip, { count: 1, start: now })
    return NextResponse.next()
  }
  if (entry.count >= RATE_LIMIT) {
    return new NextResponse("Too Many Requests", { status: 429 })
  }
  store.set(ip, { count: entry.count + 1, start: entry.start })
  return NextResponse.next()
}

export const config = {
  matcher: ['/api/:path*'],
}
"""

PHASE 10 — TROUBLESHOOTING GUIDE
Every common error you have hit across DEP I–XIV, and exactly how to fix it.

🔴 ERROR: "Can't find variable: useEffect" or "Can't find variable: useState"
Cause: The page file is using React hooks without importing them. This happens when old files from a previous session survive in your repo.

Fix: Fix: Run the Nuke script (Phase 4 Step 4.3) then the Autopatcher (Phase 4 Step 4.4). Both scripts together replace every affected file. This is the #1 error in DEP XV.

🔴 ERROR: "Database error saving new user"
Cause: This means the handle_new_user trigger is missing or broken.

Fix: Fix: Re-run the full SQL from Phase 3 Step 3.3. The trigger at the bottom of that block auto-creates a profile on signup.

🔴 ERROR: "The default export is not a React Component in /layout"
Cause: layout.js is broken or missing the RootLayout function.

Fix: Fix: Run the Autopatcher in Phase 4 Step 4.2. It completely rewrites layout.js with the correct content.

🔴 ERROR: "Module not found: @supabase/supabase-js"
Cause: The Supabase package is not installed.

Fix: Fix: cd ~/Desktop/banger-ratios && npm install @supabase/supabase-js

🔴 Terminal shows heredoc> prompt
Cause: You pressed Enter in the middle of a cat << EOF command before it was complete.

Fix: Fix: Type EOF and press Enter to escape it. Then use the Python approach from Phase 4 instead of cat heredoc commands.

🔴 ERROR: "zsh: event not found: ..."
Cause: Your terminal interpreted ! as a history expansion command.

Fix: Fix: Use single quotes around your heredoc marker (cat << 'EOF') or just use the Python approach which avoids this completely.

🔴 Leaderboard page shows nothing
Cause: Either no albums in the database, or the recalculate_banger_ratio function is missing.

Fix: Fix: (1) Add albums via Phase 8 SQL. (2) Make sure you ran the full SQL schema from Phase 3, which includes the function.

🔴 Vercel deployment fails with "Build Error"
Cause: There is a syntax error in one of your files.

Fix: Fix: Run npm run build locally. It will show you exactly which file and line number has the error. Fix that file first.

🔴 Profile photo upload fails
Cause: The avatars storage bucket is missing or its policies are not set.

Fix: Fix: Re-run the storage section of the Phase 3 SQL (the insert into storage.buckets block and the policies below it).

🔴 Artists page only shows 2 artists
Cause: Only 2 albums in your database, or all albums have the same artist.

Fix: Fix: Add more albums with different artists using the Phase 8 SQL.

🔴 News banner scrolls too fast or too slow
Cause: The animation-duration is set too low.

Fix: Fix: In NewsBanner.js the animation is 90s. To slow it down change 90s to 120s or 180s. Run the Autopatcher again and edit the 90s value before running.

🔴 Cmd+F not finding code in VS Code
Cause: You may be in the wrong file, or the variable names are different from what you expect.

Fix: Fix: Use terminal to search: grep -rn "what you are looking for" ~/Desktop/banger-ratios/src/
