-- BLOCK 1: Fix handle_new_user trigger
-- Fixes "Database error saving new user" crash on signup.
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  INSERT INTO profiles (
    id, email, username, created_at,
    banner_url, avatar_url, spotify_id, bio, display_name
  )
  VALUES (
    new.id,
    new.email,
    split_part(new.email, '@', 1),
    now(),
    null, null, null, null, null
  )
  ON CONFLICT (id) DO NOTHING;
  RETURN new;
END;
$$;
 
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
 
-- BLOCK 2: Add social profile columns
DO $$ BEGIN
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
    WHERE table_name='profiles' AND column_name='banner_url')
    THEN ALTER TABLE profiles ADD COLUMN banner_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
    WHERE table_name='profiles' AND column_name='avatar_url')
    THEN ALTER TABLE profiles ADD COLUMN avatar_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
    WHERE table_name='profiles' AND column_name='spotify_id')
    THEN ALTER TABLE profiles ADD COLUMN spotify_id text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
    WHERE table_name='profiles' AND column_name='bio')
    THEN ALTER TABLE profiles ADD COLUMN bio text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
    WHERE table_name='profiles' AND column_name='display_name')
    THEN ALTER TABLE profiles ADD COLUMN display_name text; END IF;
END $$;
 
-- BLOCK 3: follows table
CREATE TABLE IF NOT EXISTS follows (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  follower_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  following_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  created_at timestamptz DEFAULT now(),
  UNIQUE(follower_id, following_id)
);
ALTER TABLE follows ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "read follows" ON follows;
CREATE POLICY "read follows" ON follows FOR SELECT USING (true);
DROP POLICY IF EXISTS "own follows" ON follows;
CREATE POLICY "own follows" ON follows FOR ALL
  USING (auth.uid() = follower_id)
  WITH CHECK (auth.uid() = follower_id);
 
CREATE EXTENSION IF NOT EXISTS pg_trgm;
CREATE INDEX IF NOT EXISTS profiles_username_trgm
  ON profiles USING gin(username gin_trgm_ops);
 
-- BLOCK 4: Storage buckets
INSERT INTO storage.buckets (id, name, public)
  VALUES ('avatars', 'avatars', true) ON CONFLICT (id) DO NOTHING;
INSERT INTO storage.buckets (id, name, public)
  VALUES ('banners', 'banners', true) ON CONFLICT (id) DO NOTHING;
 
DROP POLICY IF EXISTS "avatar public read" ON storage.objects;
CREATE POLICY "avatar public read" ON storage.objects
  FOR SELECT USING (bucket_id = 'avatars');
DROP POLICY IF EXISTS "avatar owner write" ON storage.objects;
CREATE POLICY "avatar owner write" ON storage.objects FOR ALL
  USING (bucket_id = 'avatars' AND auth.uid()::text = (storage.foldername(name))[1]);
 
DROP POLICY IF EXISTS "banner public read" ON storage.objects;
CREATE POLICY "banner public read" ON storage.objects
  FOR SELECT USING (bucket_id = 'banners');
DROP POLICY IF EXISTS "banner owner write" ON storage.objects;
CREATE POLICY "banner owner write" ON storage.objects FOR ALL
  USING (bucket_id = 'banners' AND auth.uid()::text = (storage.foldername(name))[1]);
 
-- BLOCK 5: Verify — should show 5 tables
SELECT table_name, column_name, data_type
FROM information_schema.columns
WHERE table_schema = 'public'
AND table_name IN ('albums','tracks','ratings','profiles','follows')
ORDER BY table_name, ordinal_position;

// File: src/app/auth/page.js
'use client'
import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'
import { useRouter } from 'next/navigation'
 
function validatePassword(pw) {
  const errors = []
  if (pw.length < 8) errors.push('At least 8 characters')
  if (!/[A-Z]/.test(pw)) errors.push('One uppercase letter (A-Z)')
  if (!/[0-9]/.test(pw)) errors.push('One number (0-9)')
  if (!/[^A-Za-z0-9]/.test(pw)) errors.push('One special character (!@#$...)')
  return errors
}
 
export default function Auth() {
  const router = useRouter()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [mode, setMode] = useState('login')
  const [error, setError] = useState('')
  const [pwErrors, setPwErrors] = useState([])
 
  useEffect(() => {
    if (mode === 'signup') setPwErrors(validatePassword(password))
    else setPwErrors([])
  }, [password, mode])
 
  async function handleSubmit(e) {
    e.preventDefault()
    setError('')
    if (mode === 'signup' && pwErrors.length > 0) {
      setError('Fix password issues before continuing'); return
    }
    const { error } = mode === 'login'
      ? await supabase.auth.signInWithPassword({ email, password })
      : await supabase.auth.signUp({ email, password })
    if (error) { setError(error.message); return }
    router.push('/')
  }
 
  return (
    <div style={{ maxWidth: 400, margin: '4rem auto', padding: '2rem' }}>
      <h1>{mode === 'login' ? 'Sign In' : 'Create Account'}</h1>
      <form onSubmit={handleSubmit}>
        <input type="email" placeholder="Email" value={email}
          onChange={e => setEmail(e.target.value)} required
          style={{ display:'block', width:'100%', marginBottom:'1rem', padding:'0.5rem' }} />
        <input type="password" placeholder="Password" value={password}
          onChange={e => setPassword(e.target.value)} required
          style={{ display:'block', width:'100%', marginBottom:'0.5rem', padding:'0.5rem' }} />
        {mode === 'signup' && pwErrors.length > 0 && (
          <ul style={{ color:'#cc0000', fontSize:'0.85rem', marginBottom:'1rem' }}>
            {pwErrors.map(e => <li key={e}>{e}</li>)}
          </ul>
        )}
        {mode === 'signup' && pwErrors.length === 0 && password.length > 0 && (
          <p style={{ color:'#009944', marginBottom:'1rem' }}>✓ Password looks good</p>
        )}
        {error && <p style={{ color:'red' }}>{error}</p>}
        <button type="submit" style={{ background:'var(--pink)', color:'#fff', padding:'0.6rem 1.5rem', border:'none', cursor:'pointer', borderRadius:4 }}>
          {mode === 'login' ? 'Sign In' : 'Create Account'}
        </button>
      </form>
      <p style={{ marginTop:'1rem' }}>
        {mode === 'login'
          ? <><span>No account? </span><button onClick={() => setMode('signup')} style={{ background:'none', border:'none', color:'var(--pink)', cursor:'pointer' }}>Create one</button></>
          : <><span>Have an account? </span><button onClick={() => setMode('login')} style={{ background:'none', border:'none', color:'var(--pink)', cursor:'pointer' }}>Sign in</button></>}
      </p>
    </div>
  )
}

// File: src/app/friends/page.js
'use client'
import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'
import Link from 'next/link'
import { useRouter } from 'next/navigation'
 
export default function Friends() {
  const router = useRouter()
  const [query, setQuery] = useState('')
  const [results, setResults] = useState([])
  const [following, setFollowing] = useState(new Set())
  const [currentUser, setCurrentUser] = useState(null)
  const [feed, setFeed] = useState([])
 
  useEffect(() => {
    supabase.auth.getUser().then(({ data }) => {
      if (!data?.user) { router.push('/auth'); return }
      setCurrentUser(data.user)
      loadFollowing(data.user.id)
      loadFeed(data.user.id)
    })
  }, [])
 
  async function loadFollowing(uid) {
    const { data } = await supabase.from('follows').select('following_id').eq('follower_id', uid)
    if (data) setFollowing(new Set(data.map(r => r.following_id)))
  }
 
  async function loadFeed(uid) {
    const { data: fol } = await supabase.from('follows').select('following_id').eq('follower_id', uid)
    if (!fol?.length) return
    const ids = fol.map(r => r.following_id)
    const { data } = await supabase.from('ratings')
      .select('*, profiles(username)')
      .in('user_id', ids)
      .order('created_at', { ascending: false })
      .limit(20)
    if (data) setFeed(data)
  }
 
  async function search() {
    if (!query.trim()) return
    const { data } = await supabase.from('profiles')
      .select('id, username')
      .ilike('username', `%${query}%`)
      .limit(10)
    setResults(data || [])
  }
 
  async function toggleFollow(targetId) {
    if (!currentUser) return
    if (following.has(targetId)) {
      await supabase.from('follows').delete()
        .eq('follower_id', currentUser.id).eq('following_id', targetId)
      setFollowing(prev => { const s = new Set(prev); s.delete(targetId); return s })
    } else {
      await supabase.from('follows').insert({ follower_id: currentUser.id, following_id: targetId })
      setFollowing(prev => new Set([...prev, targetId]))
    }
  }
 
  return (
    <div style={{ maxWidth:800, margin:'2rem auto', padding:'0 1rem' }}>
      <h1>Friends</h1>
      <div style={{ display:'flex', gap:'0.5rem', marginBottom:'2rem' }}>
        <input value={query} onChange={e => setQuery(e.target.value)}
          onKeyDown={e => e.key === 'Enter' && search()}

async function uploadFile(file, bucket, suffix) {
  const ext = file.name.split('.').pop()
  const path = `${user.id}/${suffix}.${ext}`
  const { error } = await supabase.storage.from(bucket).upload(path, file, { upsert: true })
  if (error) { console.error(error); return null }
  const { data: { publicUrl } } = supabase.storage.from(bucket).getPublicUrl(path)
  return publicUrl
}
 
async function uploadBanner(file) {

  if (!url) return
  await supabase.from('profiles').update({ banner_url: url }).eq('id', user.id)
  setProfile(prev => ({ ...prev, banner_url: url }))

function truncateArtist(name, maxLen = 28) {
