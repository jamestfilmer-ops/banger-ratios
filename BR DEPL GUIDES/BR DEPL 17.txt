-- Step 10: Auto-create profile row when a new user signs up
-- This fixes the "Database error saving new user" signup bug
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer
set search_path = public as $$
begin
  insert into public.profiles (id, username, display_name)
  values (
    new.id,
    coalesce(new.raw_user_meta_data->>'username', split_part(new.email,'@',1)),
    coalesce(new.raw_user_meta_data->>'display_name', split_part(new.email,'@',1))
  )
  on conflict (id) do nothing;
  return new;
end;
$$;
 
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();
 
-- Step 11: Function to recalculate Banger Ratio after ratings are submitted
-- Called from the album page via supabase.rpc('recalculate_banger_ratio', ...)
create or replace function public.recalculate_banger_ratio(p_album_id bigint)
returns void language plpgsql security definer as $$
declare
  track_rec    record;
  total_score  numeric;
  rating_count int;
  banger_count int := 0;
  album_ratings int := 0;
begin
  -- For each track in this album, update avg_rating, total_ratings, is_banger
  -- A track is a "banger" if its average score is 5.0 or higher (out of 7)
  for track_rec in
    select id from public.tracks where album_id = p_album_id
  loop
    select count(*), coalesce(avg(score), 0)
      into rating_count, total_score
      from public.ratings
      where track_id = track_rec.id;
 
    update public.tracks set
      total_ratings = rating_count,
      avg_rating    = round(total_score::numeric, 1),
      is_banger     = (rating_count > 0 and total_score >= 5.0)
    where id = track_rec.id;
 
    if rating_count > 0 and total_score >= 5.0 then
      banger_count := banger_count + 1;
    end if;
    album_ratings := album_ratings + rating_count;
  end loop;
 
  -- Update album: total_ratings and banger_ratio (% of tracks that are bangers)
  update public.albums set
    total_ratings = album_ratings,
    banger_ratio  = case
      when (select count(*) from public.tracks where album_id = p_album_id) = 0 then 0
      else round(
        banger_count::numeric /
        (select count(*) from public.tracks where album_id = p_album_id) * 100, 1
      )
    end
  where id = p_album_id;
end;
$$;

import { useState, useEffect } from 'react'

const BADGES = {
  classic: { label: '💎 Certified Classic', min: 90 },
  gold:    { label: '🥇 Solid Gold',        min: 75 },
  hit:     { label: '🎵 Hit or Miss',       min: 60 },
  filler:  { label: '⚠️ Filler Warning',    min: 40 },
  skip:    { label: '❌ Skip It',           min: 0  },
}
 
function getBadge(ratio) {
  if (ratio >= 90) return BADGES.classic
  if (ratio >= 75) return BADGES.gold
  if (ratio >= 60) return BADGES.hit
  if (ratio >= 40) return BADGES.filler
  return BADGES.skip
}

function getBadge(r) {
  if (r >= 90) return '💎'
  if (r >= 75) return '🥇'
  if (r >= 60) return '🎵'
  return ''
}
 
export default function LeaderboardsPage() {
  const [albums, setAlbums] = useState([])
  const [genre,  setGenre]  = useState('All')
  const [decade, setDecade] = useState('All')
  const [tab,    setTab]    = useState('albums')
 
  useEffect(() => { load() }, [genre, decade])
 
  async function load() {
    let q = supabase.from('albums').select('*')
      .gt('total_ratings', 0)
      .order('banger_ratio', { ascending: false })
      .limit(50)
 
    if (genre !== 'All') q = q.eq('genre', genre)
    if (decade !== 'All') {
      const yr = parseInt(decade)
      q = q.gte('release_date', yr + '-01-01').lt('release_date', (yr + 10) + '-01-01')
    }
 
    const { data } = await q
    setAlbums(data || [])
  }
 
  // Aggregate artist data from albums
  const artistMap = {}
  albums.forEach(a => {
    if (!artistMap[a.artist_name]) artistMap[a.artist_name] = { albums: [], totalRatio: 0, count: 0 }
    artistMap[a.artist_name].albums.push(a)
    artistMap[a.artist_name].totalRatio += parseFloat(a.banger_ratio)
    artistMap[a.artist_name].count++
  })
  const artists = Object.entries(artistMap)
    .map(([name, d]) => ({ name, avgRatio: (d.totalRatio/d.count).toFixed(1), albumCount: d.count }))
    .sort((a,b) => b.avgRatio - a.avgRatio)
 
  const pill = (label, active, onClick) => (

  )

SECTION 7 — TROUBLESHOOTING
These are every error you have hit across all DEP versions, with definitive fixes.

🔴 Module not found: @/lib/supabase
Fix: The file src/lib/supabase.js is missing or in the wrong location. Run: mkdir -p ~/Desktop/banger-ratios/src/lib — then recreate the file using the code from Section 4.2.

🔴 @import rules must precede all other rules (globals.css)
Fix: The line @import "tailwindcss" must be the ABSOLUTE FIRST LINE of globals.css. If you have Google Fonts @import or any CSS before it, remove them. Load Google Fonts in layout.js instead (already done in the code in Section 5).

🔴 The default export is not a React Component
Fix: One of your page.js files is missing export default function. Check that you saved the file correctly and that the file does not have a syntax error.

🔴 Cannot read properties of undefined (reading user)
Fix: Supabase auth call failed. Check: (1) .env.local exists and has both keys, (2) you ran npm run dev not just opened the HTML file directly.

🔴 supabaseUrl is required
Fix: Your NEXT_PUBLIC_SUPABASE_URL variable is missing or empty in .env.local. Run: cat .env.local to verify both lines are present with real values (not placeholders).

🔴 Database error saving new user (on sign-up)
Fix: The trigger that auto-creates a profile row is missing. Run the full SQL block from Section 3 again — specifically the handle_new_user() function and trigger at the end.

🔴 images.domains is deprecated warning in terminal
Fix: Replace your next.config.mjs with the version in Section 4.3 which uses remotePatterns instead of domains.

🔴 Vercel deploys successfully but site shows blank white page
Fix: Your Vercel environment variables are missing. Go to Vercel → Settings → Environment Variables, add both SUPABASE keys, then Redeploy.

🔴 localhost:3000 gives 404 but vercel.com works
Fix: This is normal if the dev server is not running. Run npm run dev first. Also check you ran: cd ~/Desktop/banger-ratios before starting the server.

🔴 Rating buttons do nothing / no prompt to sign in
Fix: Check that user state is being set in album/[id]/page.js. Also check the browser console for errors (right-click → Inspect → Console tab).

🔴 Banger Ratio stays at 0% after submitting ratings
Fix: The recalculate_banger_ratio function may not be in Supabase. Run the SQL block from Section 3 again (it safely recreates the function without destroying your data if you skip the DROP TABLE lines at the top).

🔴 "Can't find variable: useEffect" or similar
Fix: A pre-existing broken file is surviving from earlier sessions. Run: rm -rf .next then npm run dev. If the error persists, the page file itself has a syntax error — compare it against the code in Section 5.

🔴 Policy already exists error in Supabase SQL
Fix: Run the DROP TABLE statements at the very top of the SQL block first, then run the full block again. The top of the block has: drop table if exists public.ratings cascade; etc.
