SECTION 1  —  WHY CMD+F IS NOT FINDING YOUR CODE — AND HOW TO FIX IT

If Cmd+F (or Ctrl+F) is not finding lines like const [seeds, setSeeds] in your .js files, there are three likely reasons:

Reason 1 — You are searching the wrong file
Make sure you have the correct file open in VS Code. Albums code lives in src/app/albums/page.js. Artists code lives in src/app/artists/page.js. The file tab at the top of VS Code shows the filename — confirm it matches.

Reason 2 — Your code uses different variable names
The guide assumes variable names like seeds and SEED_ALBUMS, but your file may use different names written during earlier sessions. Run this in terminal to see the actual variable names in your file:
# See all useState lines in albums/page.js
grep -n "useState" ~/Desktop/banger-ratios/src/app/albums/page.js
 
# See all useState lines in artists/page.js
grep -n "useState" ~/Desktop/banger-ratios/src/app/artists/page.js
 
# See the function/component names
grep -n "function\|export default" ~/Desktop/banger-ratios/src/app/albums/page.js
grep -n "function\|export default" ~/Desktop/banger-ratios/src/app/artists/page.js
The output shows every useState line with its line number. Use THOSE variable names when searching in VS Code.

Reason 3 — The component structure is different
Run this to print the first 60 lines of each file so you can see the actual structure:
head -60 ~/Desktop/banger-ratios/src/app/albums/page.js
head -60 ~/Desktop/banger-ratios/src/app/artists/page.js

✓  Once you know the real variable names and structure, Sections 2 covers adding the artwork code via terminal — no Cmd+F needed at all.

  async function uploadAvatar(file) {

    setProfile(prev => ({ ...prev, avatar_url: publicUrl }))
  }
`

  console.log('Could not find return statement')

// File: src/app/friends/page.js
'use client'
import { useState, useEffect } from 'react'
import { supabase } from '@/lib/supabase'
import Link from 'next/link'
import { useRouter } from 'next/navigation'
 
export default function Friends() {
  const router = useRouter()
  const [query, setQuery] = useState('')
  const [results, setResults] = useState([])
  const [following, setFollowing] = useState(new Set())
  const [currentUser, setCurrentUser] = useState(null)
  const [feed, setFeed] = useState([])
  const [loading, setLoading] = useState(false)
 
  useEffect(() => {
    supabase.auth.getUser().then(({ data }) => {
      if (!data?.user) { router.push('/auth'); return }
      setCurrentUser(data.user)
      loadFollowing(data.user.id)
      loadFeed(data.user.id)
    })
  }, [])
 
  async function loadFollowing(uid) {
    const { data } = await supabase.from('follows').select('following_id').eq('follower_id', uid)
    if (data) setFollowing(new Set(data.map(r => r.following_id)))
  }
 
  async function loadFeed(uid) {
    const { data: fol } = await supabase.from('follows').select('following_id').eq('follower_id', uid)
    if (!fol?.length) return
    const ids = fol.map(r => r.following_id)
    const { data } = await supabase.from('ratings')
      .select('*, profiles(username)')
      .in('user_id', ids)
      .order('created_at', { ascending: false })
      .limit(20)
    if (data) setFeed(data)
  }
 
  async function search() {
    if (!query.trim()) return
    setLoading(true)
    const { data } = await supabase.from('profiles')
      .select('id, username, avatar_url')
      .ilike('username', `%${query}%`)
      .limit(10)
    setResults(data || [])
    setLoading(false)
  }
 
  async function toggleFollow(targetId) {
    if (!currentUser) return
    if (following.has(targetId)) {
      await supabase.from('follows').delete()
        .eq('follower_id', currentUser.id).eq('following_id', targetId)
      setFollowing(prev => { const s = new Set(prev); s.delete(targetId); return s })
    } else {
      await supabase.from('follows').insert({ follower_id: currentUser.id, following_id: targetId })
      setFollowing(prev => new Set([...prev, targetId]))
    }
  }
 
  return (
    <div style={{ maxWidth:800, margin:'2rem auto', padding:'0 1rem' }}>
      <h1 style={{ color:'var(--pink)' }}>Find Friends</h1>
 
      {/* Search bar with icon button */}
      <div style={{ display:'flex', gap:'0.5rem', marginBottom:'2rem' }}>
        <input
          value={query}
          onChange={e => setQuery(e.target.value)}
          onKeyDown={e => e.key === 'Enter' && search()}
          placeholder="Search by username..."
          style={{
            flex:1, padding:'0.6rem 1rem',
            background:'#1a1a1a', border:'1px solid #333',
            color:'#fff', borderRadius:'4px 0 0 4px', fontSize:'0.95rem'
          }}
        />
        <button
          onClick={search}
          disabled={loading}
          style={{
            background:'var(--pink)', color:'#fff', border:'none',
            padding:'0.6rem 1.2rem', cursor:'pointer',
            borderRadius:'0 4px 4px 0', fontSize:'1.1rem',
            display:'flex', alignItems:'center', gap:'0.3rem'
          }}
        >
          {loading ? '…' : '🔍'}
        </button>
      </div>
 
      {/* Search results */}
      {results.length === 0 && query && !loading && (
        <p style={{ color:'#888' }}>No users found for "{query}"</p>
      )}
      {results.map(u => (
        <div key={u.id} style={{
          display:'flex', justifyContent:'space-between', alignItems:'center',
          padding:'0.75rem', border:'1px solid #222', marginBottom:'0.5rem',
          borderRadius:4, background:'#0d0d0d'
        }}>
          <div style={{ display:'flex', alignItems:'center', gap:'0.75rem' }}>
            {u.avatar_url
              ? <img src={u.avatar_url} style={{ width:36, height:36, borderRadius:'50%', objectFit:'cover' }} />
              : <div style={{ width:36, height:36, borderRadius:'50%',
                  background:'var(--pink)', display:'flex', alignItems:'center',
                  justifyContent:'center', fontWeight:700, color:'#fff' }}>
                  {u.username?.charAt(0)?.toUpperCase()}
                </div>
            }
            <Link href={'/profile/' + u.username} style={{ color:'#fff', textDecoration:'none', fontWeight:500 }}>
              {u.username}
            </Link>
          </div>
          {currentUser?.id !== u.id && (
            <button onClick={() => toggleFollow(u.id)} style={{
              background: following.has(u.id) ? '#333' : 'var(--pink)',
              color:'#fff', border:'none', padding:'0.3rem 0.9rem',
              cursor:'pointer', borderRadius:3, fontSize:'0.85rem'
            }}>
              {following.has(u.id) ? 'Unfollow' : 'Follow'}
            </button>
          )}
        </div>
      ))}
 
      {/* Friend activity feed */}
      {feed.length > 0 && (
        <div style={{ marginTop:'2rem' }}>
          <h2>Friend Activity</h2>
          {feed.map(r => (
            <div key={r.id} style={{ padding:'0.75rem', border:'1px solid #222',
              marginBottom:'0.5rem', borderRadius:4 }}>
              <span style={{ color:'var(--pink)' }}>{r.profiles?.username}</span>
              {' rated a track '}
              <strong>{r.score}/7</strong>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}

Step 1 — Add new profile columns to Supabase

-- Add settings columns to profiles table
DO $$ BEGIN
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='bio')
    THEN ALTER TABLE profiles ADD COLUMN bio text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='favorite_albums')
    THEN ALTER TABLE profiles ADD COLUMN favorite_albums jsonb DEFAULT '[]'; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='spotify_url')
    THEN ALTER TABLE profiles ADD COLUMN spotify_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='apple_music_url')
    THEN ALTER TABLE profiles ADD COLUMN apple_music_url text; END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='profiles' AND column_name='display_name')
    THEN ALTER TABLE profiles ADD COLUMN display_name text; END IF;
END $$;
 
-- Allow users to update their own profile
DROP POLICY IF EXISTS "update own profile" ON profiles;
CREATE POLICY "update own profile" ON profiles
  FOR UPDATE USING (auth.uid() = id);

// File: src/middleware.js
import { NextResponse } from 'next/server'
 
// Simple in-memory rate limiter (resets on cold start)
// For production, use Upstash Redis for persistent limits
const rateMap = new Map()
 
export function middleware(request) {
  const ip = request.headers.get('x-forwarded-for') || 'unknown'
  const now = Date.now()
  const windowMs = 60 * 1000  // 1 minute
  const maxRequests = 60      // 60 requests per minute per IP
 
  const entry = rateMap.get(ip) || { count: 0, start: now }
 
  if (now - entry.start > windowMs) {
    entry.count = 1
    entry.start = now
  } else {
    entry.count++
  }
  rateMap.set(ip, entry)
 
  // Clean old entries occasionally
  if (rateMap.size > 5000) {
    for (const [k, v] of rateMap) {
      if (now - v.start > windowMs * 2) rateMap.delete(k)
    }
  }
 
  if (entry.count > maxRequests) {
    return new NextResponse('Too many requests', {
      status: 429,
      headers: { 'Retry-After': '60' }
    })
  }
 
  return NextResponse.next()
}
 
export const config = {
  matcher: ['/api/:path*', '/auth/:path*'],
}

SECTION 13  —  THE HEREDOC> PROBLEM — WHAT IT IS AND HOW TO ESCAPE

heredoc> is NOT an error. It means your terminal is waiting for you to finish typing a multi-line command. It appears when you used << 'HEREDOC' (or << EOF) and the terminal has not seen the closing HEREDOC or EOF word yet.

Why it happens
The cat << 'HEREDOC' pattern tells your shell "everything until you see the word HEREDOC on its own line is the file content." If the closing HEREDOC word is missing, mistyped, or has a space before it, the terminal keeps waiting.

How to escape from heredoc>
You see heredoc>	Type the closing word on its own line and press Enter
Closing word was HEREDOC	Type:   HEREDOC   then Enter
Closing word was EOF	Type:   EOF   then Enter
Nothing works	Press Ctrl+C to cancel the whole command

# Example — if you see heredoc> and your closing word is HEREDOC:
HEREDOC
 
# Or press Ctrl+C to cancel entirely and start over

How to avoid it
•	Copy-paste the ENTIRE command block at once, not line by line
•	Make sure the closing word (HEREDOC or EOF) appears on its very own line with NO spaces before or after it
•	The opening and closing word must match exactly — same capitalisation
•	If you are on a Mac, use Terminal.app or iTerm2, not the VS Code integrated terminal for heredoc commands — VS Code terminal sometimes mishandles paste formatting

Testing that your heredoc worked
# After running a heredoc command, always verify the file was created:
ls -la src/app/friends/page.js
 
# Check the first few lines look right:
head -5 src/app/friends/page.js
